3ds Verification​

  • Using the session vault token or permanent vault token. Perform a 3ds pre auth request.

POST /v1/charges/3ds HTTP/1.1
Host: api.paydock.com
x-user-public-key: {{public-key}}
Content-Type: application/json

{
    "amount": "103",
    "currency": "AUD",
    "customer": {
        "payment_source": {
            "vault_token": vault_token,
            "gateway_id": "gateway-id"
        } 
},
    "_3ds": {
        "browser_details": {
            "name": "CHROME",
            "java_enabled": "true",
            "language": "en-US",
            "screen_height": "640",
            "screen_width": "480",
            "time_zone": "273",
            "color_depth": "24"
        } 
    }
}
  • Sample response.

{
    "status": 201,
    "error": null,
    "resource": {
        "type": "charge",
        "data": {

  • Using the resource.data._3ds. Create a Canvas3ds widget using the client-sdk.

    <div id="widget"></div>
    <script src="https://widget.paydock.com/sdk/latest/widget.umd.min.js"></script>

        var widget = new paydock.Canvas3ds('#widget', token);
        widget.setEnv('sandbox');
        widget.on('chargeAuthSuccess', function (data) {
            console.log(data);
        });
        widget.on('chargeAuthReject', function (data) {
            console.log(data);
        });
        widget.on('chargeAuthCancelled', function (data) {
            console.log(data);
        });
        widget.on('additionalDataCollectSuccess', function (data) {
            console.log(data);
        });
        widget.on('additionalDataCollectReject', function (data) {
            console.log(data);
        });
        widget.on('chargeAuth', function (data) {
            console.log(data);
        });
        widget.load();
    </script>

Perform a direct charge with the customer_id​


POST /v1/charges HTTP/1.1
Host: api.paydock.com
x-user-secret-key: {{secret-key}}
Content-Type: application/json

{
	"amount":"100",
	"currency":"AUD",
	"customer_id":"5e4bfbd86d244b438451fbd7"
}

x-user-secret-key: {{secret-key}}
Content-Type: application/json

{
    "amount": "10.00",
    "currency": "AUD",
    "customer_id": customer_id,
    "customer": {
        "payment_source": {
            "vault_token": vault_token,
            "gateway_id": gateway_id
        }
    }
